var countBits = function(n) {
    // n &= (n-1)
    let bitArr = [];

    const countBit = (i) => {
        let num = 0;
        while(i > 0) {
            num += 1;
            i &= (i-1);
        }
        return num;
    }
    
    for(let i = 0; i <= n; i++) {
        bitArr[i] = countBit(i);
    }
    return bitArr;

};
