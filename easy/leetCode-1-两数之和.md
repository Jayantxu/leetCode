var twoSum = function(nums, target) {
    let idxMap = new Map();
    let startIdx = '', count = 0;
    for(let iTem of nums) {
        let diff = target - iTem;
        if(idxMap.has(diff)) {
            startIdx = idxMap.get(diff);
        } else {
            idxMap.set(iTem, count);
            count++;
        }
        if(startIdx !== '') {
            return [startIdx, count]
        }
    }
};
