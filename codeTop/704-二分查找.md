var search = function(nums, target) {
    let L = 0,
        R = nums.length - 1;
    let mid = Math.floor((L + R) / 2);
    while(L <= R) {
        if(nums[mid] > target) {
            R = mid - 1;
        } else if (nums[mid] < target) {
            L = mid + 1;
        } else if (nums[mid] === target) {
            return mid;
        }
        mid = Math.floor((L + R) / 2);
    }
    return -1;
};
