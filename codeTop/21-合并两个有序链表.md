var mergeTwoLists = function(l1, l2) {
    if(!l1) return l2;
    if(!l2) return l1;
    let resList = new ListNode(null);
    let tempList = resList;
    while(l1 && l2) {
        if(l1.val < l2.val) {
            tempList.next = l1;
            l1 = l1.next;
        } else {
            tempList.next = l2;
            l2 = l2.next;
        }
        tempList = tempList.next;
    }
    if(l1) {
        tempList.next = l1;
    }
    if(l2) {
        tempList.next = l2;
    }
    return resList.next;
};
