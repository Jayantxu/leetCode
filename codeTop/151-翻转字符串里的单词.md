var reverseWords = function(s) {
    s = s.trim();
    s = s.split(' ');
    s.reverse();
    let i = 0;
    let resArr = []
    while(i < s.length) {
        if(s[i]) {
            resArr.push(s[i])
        }
        i+=1;
    }
    return resArr.join(' ');
};
