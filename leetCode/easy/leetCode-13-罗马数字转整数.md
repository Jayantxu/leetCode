var romanToInt = function(s) {
    let strMap = new Map();
    strMap.set('I', 1);
    strMap.set('V', 5);
    strMap.set('X', 10);
    strMap.set('L', 50);
    strMap.set('C', 100);
    strMap.set('D', 500);
    strMap.set('M', 1000);
    strMap.set('IV', 4);
    strMap.set('IX', 9);
    strMap.set('XL', 40);
    strMap.set('XC', 90);
    strMap.set('CD', 400);
    strMap.set('CM', 900);

    let strLen = s.length;
    let num = 0;
    for(let i = strLen - 1; i >= 0; ) {
        if( strMap.has(s.slice(i - 1, i+ 1)) ) {
            num += strMap.get(s.slice(i - 1, i + 1));
            i -= 2;
        } else {
            num += strMap.get(s.slice(i, i + 1));
            i -= 1;
        }
    }
    return num;
};
