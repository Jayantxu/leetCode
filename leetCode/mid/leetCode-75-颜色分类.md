var sortColors = function(nums) {
    let nLen = nums.length;
    let flag = 0;
    while(nums[flag] === 0) {
        flag += 1;
    }
    let iStart = flag;
    for(let i = iStart; i < nLen; i++) {
        if(nums[i] === 0) {
            [nums[i], nums[flag]] = [nums[flag], nums[i]];
            flag += 1;
        }
        while(nums[flag] === 0) {
            flag += 1;
        }
    }
    iStart = flag;
    flag = nLen - 1;
    while(nums[flag] === 2) {
        flag -= 1;
    }
    for(let i = iStart; i < flag; i++) {
        if(nums[i] === 2) {
            [nums[i], nums[flag]] = [nums[flag], nums[i]];
            flag -= 1;
        }
        while(nums[flag] === 2) {
            flag -= 1;
        }
    }
    return nums;
};
