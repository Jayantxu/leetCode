var nextPermutation = function(nums) {
    let idx = nums.length - 2 ;
    while(idx >= 0 && nums[idx] >= nums[idx + 1]) {
        idx--;
    }
    if(idx >= 0) {
        let jdx = nums.length - 1;
        while(jdx >= 0 && nums[idx] >= nums[jdx]) {
            jdx--;
        }
        [nums[idx], nums[jdx]] = [nums[jdx], nums[idx]];
    }
    idx = idx + 1;
    let jdx = nums.length - 1;
    while(idx < jdx) {
        [nums[idx], nums[jdx]] = [nums[jdx], nums[idx]];
        idx++;
        jdx--;
    }
};
