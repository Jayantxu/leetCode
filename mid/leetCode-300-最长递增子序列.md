var lengthOfLIS = function(nums) {
    if(!nums.length) return 0;
    
    let dp = new Array(nums.length);
    
    dp[0] = 1;
    let recordMax = 1;
    for(let i = 1; i < nums.length; i++) {
        dp[i] = 1;
        for(let j = 0; j < i; j++) {
            if(nums[i] > nums[j]) {
                dp[i] = Math.max(dp[i], dp[j] + 1);
            }
        }
        recordMax = Math.max(recordMax, dp[i]);
    }
    return recordMax;
};
