var detectCycle = function(head) {
    if(!head) return null;
    if(!head.next) return null;
    let p1 = head,
        p2 = head.next;
    while(p1 !== p2) {
        if(!p2) return null;
        if(!p2.next) return null;
        p1 = p1.next;
        p2 = p2.next.next;
    }
    p2 = head;
    while(p1 !== p2) {
        p1 = p1.next;
        p2 = p2.next;
    }
    return p1;
};
