var maxDepth = function(root) {
    let ht = 0
    const dfs = (Root, tmpH) => {
        if(!Root) return ;
        ht = Math.max(ht, tmpH);
        dfs(Root.left, tmpH + 1);
        dfs(Root.right, tmpH + 1);
    }
    dfs(root, 1);
    return ht;
};
