var subsets = function(nums) {
    let nLen = nums.length;
    let resArr = [];
    const numfs = (idx, arr) => {
        resArr.push(arr.slice());
        for(let i = idx; i < nLen; i++) {
            arr.push(nums[i]);
            numfs(i + 1, arr);
            arr.pop();
        }
    }
    numfs(0, []);
    return resArr;
};
