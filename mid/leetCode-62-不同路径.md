var uniquePaths = function(m, n) {
    const pathNumArr = new Array(m).fill(0).map(() => new Array(n).fill(0));
    for (let i = 0; i < m; i++) {
        pathNumArr[i][0] = 1;
    }
    for (let j = 0; j < n; j++) {
        pathNumArr[0][j] = 1;
    }
    for (let i = 1; i < m; i++) {
        for (let j = 1; j < n; j++) {
            pathNumArr[i][j] = pathNumArr[i - 1][j] + pathNumArr[i][j - 1];
        }
    }
    return pathNumArr[m - 1][n - 1];
};
