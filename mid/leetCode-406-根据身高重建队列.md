var reconstructQueue = function(people) {
    people.sort((a, b) => b[0] - a[0] || a[1] - b[1]);
    let resArr = [];
    for(let i = 0; i < people.length; i++) {
        let nowMan = people[i];
        let resLen = resArr.length;
        if(resLen > nowMan[1]) {
            let nextArr = resArr.splice(nowMan[1]);
            resArr.push(nowMan);
            resArr = resArr.concat(nextArr);
        } else {
            resArr.push(nowMan);
        }
    }
    return resArr;
};
