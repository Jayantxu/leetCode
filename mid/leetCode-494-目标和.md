var findTargetSumWays = function(nums, target) {
    let resNum = 0;
    const dfsNum = (idx, nextTarget) => {
        if(idx >= nums.length) {
            if(nextTarget === target) {
                resNum+=1;
            }
        } else {
            dfsNum(idx + 1, nextTarget + nums[idx]);
            dfsNum(idx + 1, nextTarget - nums[idx]);
        }
    };
    dfsNum(0, 0);
    return resNum;
};
