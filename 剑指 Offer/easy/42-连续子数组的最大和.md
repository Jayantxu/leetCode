var maxSubArray = function(nums) {
    if(!nums.length) return;
    let recordMax = nums[0];
    let maxArr = [nums[0]];
    for(let i = 1; i < nums.length; i++) {
        maxArr[i] = Math.max(maxArr[i - 1] + nums[i], nums[i]);
        recordMax = Math.max(recordMax, maxArr[i]);
    }
    return recordMax;
};
