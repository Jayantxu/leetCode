/**
 * // Definition for a Node.
 * function Node(val,left,right) {
 *    this.val = val;
 *    this.left = left;
 *    this.right = right;
 * };
 */
/**
 * @param {Node} root
 * @return {Node}
 */
var treeToDoublyList = function(root) {
    
    if(!root) return;
    let head = null,
        pre = head;
    inOrderRoot(root);
    head.left = pre;
    pre.right = head;
    function inOrderRoot(node) {
        if(!node) return ;
        inOrderRoot(node.left, pre)
        if(!pre) {
            head = node;
        } else {
            pre.right = node;
        }
        node.left = pre;
        pre = node;
        inOrderRoot(node.right, pre)
    }
    return head;
};
