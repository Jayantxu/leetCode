var myPow = function(x, n) {
    if(n === 0) {
        return 1;
    }
    unit = x;
    if(n < 0) {
        x = 1 / x;
        n = Math.abs(n);
        unit = x;
    }
    let i = 1;
    while(i < n ) {
        x *= unit;
        i += 1;
    }
    return x;
};
