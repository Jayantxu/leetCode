/**
 * @param {number[][]} matrix
 * @param {number} target
 * @return {boolean}
 */
var findNumberIn2DArray = function(matrix, target) {
    if(!matrix.length) {
        return false;
    }
    let mLen = matrix.length,
        nLne = matrix[0].length;
    let i = 0,
        j = nLne - 1;
    while(i < mLen && j >= 0) {
        if(matrix[i][j] === target) {
            return true;
        } else if (matrix[i][j] < target) {
            i += 1;
        } else if (matrix[i][j] > target) {
            j -= 1;
        }
    }
    return false;
};
